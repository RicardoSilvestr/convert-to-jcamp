convert-to-jcamp

convert-to-jcamp

Convert strings into JCAMP.

Maintained by Zakodium

Installation

$ npm install convert-to-jcamp

Usage

From Variables

 
const { fromVariables } = require('convert-to-jcamp');
const variables = {
  x: {
    data: [1, 2, 3, 4],
    label: 'x value',
    units: 'x unit',
    isDependent: false,
  },
  y: { data: [2, 3, 4, 5], label: 'y value', units: 'y unit' },
};

const jcamp = fromVariables(variables, {
  forceNtuples: true,
  meta: {
    meta1: 'value1',
    meta2: 'value2',
  },
  info: {
    title: 'Hello world',
    dataType: 'TEST',
  },
});
